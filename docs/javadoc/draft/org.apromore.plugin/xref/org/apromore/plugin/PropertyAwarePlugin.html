17  package org.apromore.plugin;
18  
19  import java.util.Set;
20  
21  import org.apromore.plugin.property.PropertyType;
22  
23  /**
24   * A PropertyAwarePlugin defines a Set of properties that may or must be provided by the caller for proper operation of the Plugin.
25   *
26   * @author <a href="mailto:felix.mannhardt@smail.wir.h-brs.de">Felix Mannhardt (Bonn-Rhein-Sieg University oAS)</a>
27   *
28   */
29  public interface PropertyAwarePlugin extends Plugin {
30  
31      /**
32       * Returns the Set of all available properties for this Plugin.
33       *
34       * @return Set of PropertyType that contains both mandatory and optional properties.
35       */
36      Set<PropertyType<?>> getAvailableProperties();
37  
38      /**
39       * Returns the Set of all required properties for this Plugin.
40       *
41       * @return Set of PropertyType that are mandatory
42       */
43      Set<PropertyType<?>> getMandatoryProperties();
44  
45      /**
46       * Returns the Set of all optional properties for this Plugin. That should be the result of {@link #getAvailableProperties()} minus the result of
47       * {@link #getMandatoryProperties()}.
48       *
49       * @return Set of PropertyType that are optional
50       */
51      Set<PropertyType<?>> getOptionalProperties();
52  
53  }
