17  package org.apromore.canoniser;
18  
19  import java.io.InputStream;
20  import java.io.OutputStream;
21  
22  import javax.xml.bind.JAXBElement;
23  import javax.xml.bind.JAXBException;
24  
25  import org.apromore.anf.ANFSchema;
26  import org.apromore.anf.AnnotationsType;
27  import org.apromore.cpf.CPFSchema;
28  import org.apromore.cpf.CanonicalProcessType;
29  import org.apromore.plugin.impl.DefaultPropertyAwarePlugin;
30  import org.xml.sax.SAXException;
31  
32  /**
33   * Implements common functionality shared by all Canonisers and reads the supported native type from the Canonisers 'plugin.config' file. The key used
34   * is: 'canoniser.nativeType'.
35   *
36   * @author <a href="mailto:felix.mannhardt@smail.wir.h-brs.de">Felix Mannhardt (Bonn-Rhein-Sieg University oAS)</a>
37   *
38   */
39  public abstract class DefaultAbstractCanoniser extends DefaultPropertyAwarePlugin implements Canoniser {
40  
41      /*
42       * (non-Javadoc)
43       *
44       * @see org.apromore.canoniser.Canoniser#getNativeType()
45       */
46      @Override
47      public String getNativeType() {
48          return getConfigurationByName("canoniser.nativeType");
49      }
50  
51      /**
52       * Marshal the Canonical Process Format into the provided OutputStream with Validation.
53       *
54       * @param canonicalFormat
55       * @param cpf
56       * @throws JAXBException
57       * @throws SAXException
58       */
59      protected void marshalCanoncialFormat(final OutputStream canonicalFormat, final CanonicalProcessType cpf) throws JAXBException, SAXException {
60          CPFSchema.marshalCanoncialFormat(canonicalFormat, cpf, true);
61      }
62  
63      /**
64       * Marshal the Annotations Format into the provided OutputStream with Validation.
65       *
66       * @param annotationFormat
67       * @param anf
68       * @throws JAXBException
69       * @throws SAXException
70       */
71      protected void marshalAnnotationFormat(final OutputStream annotationFormat, final AnnotationsType anf) throws JAXBException, SAXException {
72          ANFSchema.marshalAnnotationFormat(annotationFormat, anf, true);
73      }
74  
75      /**
76       * Unmarshals the Canonical Process Format from the provided InputStream with Validation.
77       *
78       * @param canonicalFormat
79       * @return JAXBElement holding CanonicalProcessType
80       * @throws JAXBException
81       * @throws SAXException
82       */
83      protected JAXBElement<CanonicalProcessType> unmarshalCanonicalFormat(final InputStream canonicalFormat) throws JAXBException, SAXException {
84          return CPFSchema.unmarshalCanonicalFormat(canonicalFormat, true);
85      }
86  
87      /**
88       * Unmarshal the Annotations Format from the provided InputStream without Validation.
89       *
90       * @param annotationsFormat
91       * @return
92       * @throws JAXBException
93       * @throws SAXException
94       */
95      protected JAXBElement<AnnotationsType> unmarshalAnnotationFormat(final InputStream annotationFormat) throws JAXBException, SAXException {
96          if (annotationFormat != null) {
97              return ANFSchema.unmarshalAnnotationFormat(annotationFormat, false);
98          } else {
99              return null;
100         }
101     }
