22  /**
23   * Interface for Deployment Plugins, that support deploying processes to a process/work-flow engine.
24   *
25   * @author <a href="mailto:felix.mannhardt@smail.wir.h-brs.de">Felix Mannhardt</a>
26   *
27   */
28  public interface DeploymentPlugin extends PropertyAwarePlugin {
29  
30      /**
31       * Native type the Deployment Plugin uses
32       *
33       * @return native type the process will be deployed to
34       */
35      String getNativeType();
36  
37  	/**
38  	 * Deploys the process in canonical process format to a process/work-flow engine.
39  	 *
40  	 * @param canonicalProcess the process to be deployed
41       * @throws DeploymentException in case of an error during deployment
42  	 * @throws PluginException in case of an error related to Plugins
43  	 */
44  	PluginResult deployProcess(CanonicalProcessType canonicalProcess, PluginRequest request) throws DeploymentException, PluginException;
45  
46  	/**
47       * Deploys the process in canonical process format to a process/work-flow engine.
48       *
49       * @param canonicalProcess the process to be deployed
50       * @param annotation to be used during deployment
51       * @throws DeploymentException in case of an error during deployment
52  	 * @throws PluginException in case of an error related to Plugins
53       */
54  	PluginResult deployProcess(CanonicalProcessType canonicalProcess, AnnotationsType annotation, PluginRequest request) throws DeploymentException, PluginException;
55  
56  }
